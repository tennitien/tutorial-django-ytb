```python
from django.views import View
from django.contrib.auth import authenticate,login, decorators
from django.contrib.auth.mixins import LoginRequiredMixin

# Cach 1:
class ViewUser(View):
    def get(self,req):
        if not req.user.is_authenticated:
            return HttpResponse('Vui long dang nhap')
        else:
            return HttpResponse('Ban da dang nhap r')

# Cach 2: ngan hon

@decorators.login_required(login_url='/login/')
def view_product(req):
    return HttpResponse('decor')

# Cach 3: tot nhat
# thu tu : LoginRequiredMixin, View --> de tranh loi
class LoginRequire(LoginRequiredMixin, View):
    login_url='/login/'
    def get(self,req):
        return HttpResponse('Ban da dang nhap r')
```

- urls:

```python

urlpatterns = [
```
